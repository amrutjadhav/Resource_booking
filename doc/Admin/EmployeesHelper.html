def list_employees
	employees = current_employee.company.employees
	@employees = []
	employees.each do | emp |
		@employees << [emp.name, emp.id]
	end
end

def list_roles
	roles = current_employee.company.roles
	@roles = []
	roles.each do | role |
		if role.designation!="admin"
			@roles << [role.designation, role.id]
		end
	end
end
